### Creating db and tables

`create database SchoolDB;`

`use SchoolDB;`

creating Students table
>
    create table Students(
        student_id int primary key,
        name varchar(50),
        age int,
        grade varchar(10)
    );

### Questions
---
1. Select all records from 'Students'.
>
    select * 
    from Students;

---
2. Select names and grades of students only.
>
    select name,grade 
    from Students;

---
3. Select students who are older than 14 years.
>
    select name 
    from Students 
    where age > 14;

---
4. Select students in '8th' grade.
>
    select * 
    from Students 
    where grade = "8th";

---
5. Update Alice age from 14 to 15.
>
    update Students 
    set age = 15 
    where name = "Alice";

---
6. Delete a student record with student_id = 3.
>
    delete 
    from Students 
    where student_id = 3;

---
7. Delete all students in '7th' grade.
>
    delete 
    from Students 
    where grade = "7th";

---
8. Write an SQL query to drop the 'Students' table from the database. 
>
    drop table Students;
